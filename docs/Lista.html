def extraer_delante
    if(@tam == 0)
       puts "La Lista está vacía"
    else
       aux = @cabeza
       @cabeza = @cabeza[:Next_]
       @cabeza[:prev] = nil
       @tam = @tam - 1
       return aux[:value]
    end
end

def insertar_delante(dato)
    dato_ = Nodo.new(dato,nil,nil)
    if (tam == 0) 
        @cola = dato_ 
        @cabeza = dato_
        
    else  
        dato_[:Next_] = @cabeza  #|nil|cabeza/dato|nil
        @cabeza.prev = dato_
        dato_[:prev] = nil
        @cabeza = dato_
        
    end 
    @tam = @tam + 1

end

def insertar_detras(dato)
    dato_ = Nodo.new(dato,nil,nil)
    if (@tam == 0)
        @cabeza = dato_
        @cola = dato_
    else
        dato_[:prev] = @cola
        @cola[:Next_]= dato_
        dato_[:Next_] = nil
        @cola = dato_
    end
   
    @tam = @tam + 1 
end
